No telemetry yet. Configure your agent to send OTel data to this endpoint:

OTLP/HTTP: http://localhost:14318/v1/otlp

# Claude Code
export CLAUDE_CODE_ENABLE_TELEMETRY=1
export OTEL_METRICS_EXPORTER=otlp
export OTEL_LOGS_EXPORTER=otlp
export OTEL_EXPORTER_OTLP_PROTOCOL=http/protobuf
export OTEL_EXPORTER_OTLP_ENDPOINT=http://localhost:14318

# OpenClaw
openclaw config set diagnostics.otel.endpoint http://localhost:14318/v1/otlp
openclaw config set diagnostics.otel.traces true
openclaw config set diagnostics.otel.metrics true
openclaw config set diagnostics.otel.logs true # optional: gateway internal logs

# Any OTel SDK
OTEL_EXPORTER_OTLP_ENDPOINT=http://localhost:14318 your-agent
